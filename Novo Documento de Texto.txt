require('dotenv').config(); // Carrega as variáveis do arquivo .env
const { Client, GatewayIntentBits, EmbedBuilder } = require('discord.js');
const axios = require('axios');

// --- CONFIGURAÇÕES ---
const TOKEN = process.env.DISCORD_TOKEN; 
const API_URL = 'https://mbappeout.replit.app/api/votes';

const client = new Client({
    intents: [
        GatewayIntentBits.Guilds,
        GatewayIntentBits.GuildMessages,
        GatewayIntentBits.MessageContent
    ]
});

client.once('ready', () => {
    console.log(`✅ Bot online as ${client.user.tag}!`);
});

client.on('messageCreate', async (message) => {
    if (message.author.bot || !message.content.startsWith('!')) return;

    if (message.content.toLowerCase() === '!vote') {
        try {
            const response = await axios.post(API_URL, "11", {
                headers: {
                    'Content-Type': 'text/plain',
                    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) Chrome/146.0.0.0'
                }
            });

            const count = response.data.count;

            const voteEmbed = new EmbedBuilder()
                .setColor(0x5865F2)
                .setTitle('✅ Vote Success!')
                .setDescription('Your vote for **11** has been registered.')
                .addFields({ 
                    name: 'Current Total Votes', 
                    value: `**${count.toLocaleString('en-US')}**`, 
                    inline: false 
                })
                .setFooter({ text: 'Game API System' })
                .setTimestamp();

            await message.reply({ embeds: [voteEmbed] });

        } catch (error) {
            console.error('Error:', error.message);
            await message.reply('❌ Failed to connect to the game server.');
        }
    }
});

client.login(TOKEN);
